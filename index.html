Static typing

Variables 
							
								let isAdmin: boolean = true;
Functions 
							
								function greet(name: string): string {
									return `Hello, ${name}!`;

Basic types

Primitive types 
							
								let age: number = 30;
								let name: string = "Alice";
								let isAdmin: boolean = true;
								// ...bigInt, symbol, undefined, null.

Arrays

Array of numbers and tuples 
let numbers: number[] = [1, 2, 3];
let coordinates: [number, number] = [10, 20];
let person: [string, number, boolean] = ["Alice", 30, true];

Objects

let person: { 
	name: string; 
	age: number;
} = { 
	name: "Tom", 
	age: 23 
};

Interfaces

interface User { 
	id: number;
	name: string;
	age?: number;
}

const user: User = { 
	id: 1, 
	name: "Alice", 
	age: 30, 
};

Type aliases

type Id = number | string | boolean;  

let userId: Id = 2; // number  

userId = "qwerty"; // string
userId = true; // boolean

Functions

function add(a: number, b: number): number { 
	return a + b; 
}
( ) => void 
function logMessage(message: string): void { 
	console.log(message); 
}
( ) => never 
function throwError(message: string): never { 
	throw new Error(message); 
}

Generics

function identity<T>(arg: T): T { 
	return arg; 
} 

const result1 = identity(42); // number
const result2 = identity("Hello"); // string
						
class Box<T> { 
	private value: T; 
	constructor(value: T) { 
		this.value = value; 
	} 

	getValue(): T { 
		return this.value; 
	}
} 

const box1 = new Box(42); // number 
const box2 = new Box("Hello"); // string

any

Function without specified types... 
function sum(a, b) { 
	return a + b;
}
...is the same as 
function sum(a: any, b: any): any { 
	return a + b;
}
Sometimes this type may be necessary. 
let data: any = JSON.parse('{ "id": "abc" }'); 
let id = data.id;

Avoid it if possible!
